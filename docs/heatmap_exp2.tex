% ============================================================
%  heatmap_exp2.tex
%
%  Stand-alone compilable figure: FIFO vs RVI routing heatmap
%  for Experiment 2 (asymmetric costs, 2 fully flexible servers).
%
%  Configuration
%    tick_rate = 3,  lambda = (0.25, 0.25),  mu = 0.35
%    c = (100, 300),  D_0 = 3,  D_1 = 3  (physical time units)
%    => D0_ticks = D1_ticks = 9  (both deadlines at FIL=9)
%
%  Axis orientation:
%    X-axis: FIL_1  (left=0, right=12)
%    Y-axis: FIL_0  (bottom=0, top=12)   <- natural
%
%  RVI policy decoding (canonical state: pool0 busy on type0, pool1 idle):
%    "." on/below diagonal (FIL_0 >= FIL_1, FIL_1 > 0):
%         skip type-0 (FIFO-top) at action alpha=0;
%         next action (alpha=1) assigns type-1 to pool1  ->  orange
%    "." with FIL_1 = 0:
%         no type-1 waiting; pool1 stays idle  ->  idle (shown orange, covered by alpha)
%    "1" above diagonal (FIL_1 > FIL_0):
%         direct assign type-1 to pool1  ->  orange
%    "0" at (FIL_0=10, FIL_1=0):
%         type-0 has exceeded deadline (FIL_0>9), type-1 just arrived;
%         assign type-0 urgently  ->  BLUE  [unique exception]
%
%  Result: RVI is uniformly orange except the single blue dot at (FIL_0=10, FIL_1=0).
%  FIFO/RVI = 1.73 (72.6% cost gap); FIFO*Lambda=70.38, RVI*Lambda=40.79.
% ============================================================
\documentclass[11pt,a4paper]{article}
\usepackage{tikz}
\usepackage{amsmath}
\usepackage[margin=2cm]{geometry}
\begin{document}

\begin{figure}[htbp]
\centering
\begin{tikzpicture}[x=0.46cm, y=0.46cm]

  % ============================================================
  % LEFT PANEL — FIFO policy
  %
  % Always serves the oldest waiting job (highest FIL).
  %   FIL_0 >= FIL_1  =>  type 0 is FIFO-top  =>  blue
  %   FIL_1  > FIL_0  =>  type 1 is FIFO-top  =>  orange
  % ============================================================
  \begin{scope}

    \foreach \i in {0,...,12}{       % \i = FIL_0 (row, y-axis, 0 at bottom)
      \foreach \j in {0,...,12}{     % \j = FIL_1 (col, x-axis, 0 at left)
        \ifnum\i<\j
          \fill[orange!65] (\j,\i) rectangle (\j+1,\i+1);
        \else
          \fill[blue!55]   (\j,\i) rectangle (\j+1,\i+1);
        \fi
      }
    }

    \draw[gray!45, very thin] (0,0) grid (13,13);

    % Type-1 deadline: cost fires for FIL_1 > 9  (vertical)
    \draw[red!70, dashed, thick] (10,0) -- (10,13);
    \node[red!70, font=\tiny, above] at (10,13.5) {$d_1{=}9$};

    % Type-0 deadline: cost fires for FIL_0 > 9  (horizontal)
    \draw[red!70, dashed, thick] (0,10) -- (13,10);
    \node[red!70, font=\tiny, right] at (13.2,10.5) {$d_0{=}9$};

    \foreach \j in {0,3,6,9,12}
      \node[below, font=\scriptsize] at (\j+0.5,-0.3) {\j};
    \foreach \i in {0,3,6,9,12}
      \node[left, font=\scriptsize] at (-0.3,\i+0.5) {\i};

    \node[below=5pt, font=\small] at (6.5,-1.1) {$\mathrm{FIL}_1$};
    \node[rotate=90, font=\small] at (-1.6,6.5) {$\mathrm{FIL}_0$};

    \draw[black, thin] (0,0) rectangle (13,13);
    \node[font=\small\bfseries, above=3pt] at (6.5,13) {FIFO};

  \end{scope}

  % ============================================================
  % RIGHT PANEL — RVI optimal policy
  %
  % After fixing the complete_job bug (gang-release -> per-pool decrement)
  % the optimal policy collapses to: always serve type 1 (orange),
  % with one single exception.
  %
  % Decoded full output (13x13 canonical grid):
  %   All cells: orange (type-1 served, via direct assign or skip-then-assign chain)
  %   Exception: (FIL_1=0, FIL_0=10) -> blue (type-0 past deadline, type-1 just arrived)
  % ============================================================
  \begin{scope}[xshift=7.5cm]

    % Background: type 1 served everywhere (direct or via skip-then-assign)
    \fill[orange!65] (0,0) rectangle (13,13);

    % Single blue cell at TikZ (x=FIL_1=0, y=FIL_0=10):
    %   FIL_0=10 > d_0=9  =>  type-0 is accruing deadline penalty
    %   FIL_1=0            =>  type-1 just arrived, no urgency
    %   RVI: serve type-0 to stop the ongoing cost.
    \fill[blue!55] (0,10) rectangle (1,11);

    \draw[gray!45, very thin] (0,0) grid (13,13);

    % Deadline lines (same as FIFO panel)
    \draw[red!70, dashed, thick] (10,0) -- (10,13);
    \node[red!70, font=\tiny, above] at (10,13.5) {$d_1{=}9$};

    \draw[red!70, dashed, thick] (0,10) -- (13,10);
    \node[red!70, font=\tiny, right] at (13.2,10.5) {$d_0{=}9$};

    \foreach \j in {0,3,6,9,12}
      \node[below, font=\scriptsize] at (\j+0.5,-0.3) {\j};
    \node[below=5pt, font=\small] at (6.5,-1.1) {$\mathrm{FIL}_1$};

    \draw[black, thin] (0,0) rectangle (13,13);
    \node[font=\small\bfseries, above=3pt] at (6.5,13) {RVI (optimal)};

  \end{scope}

  % ============================================================
  % Legend
  % ============================================================
  \begin{scope}[yshift=-2.2cm]
    \fill[blue!55]   (0cm,0cm) rectangle (0.46cm,0.46cm);
    \node[right, font=\footnotesize] at (0.56cm,0.23cm) {type~0 served};

    \fill[orange!65] (3.8cm,0cm) rectangle (4.26cm,0.46cm);
    \node[right, font=\footnotesize] at (4.36cm,0.23cm) {type~1 served};

    \draw[red!70, dashed, thick] (8.6cm,0.23cm) -- (9.1cm,0.23cm);
    \node[right, font=\footnotesize] at (9.2cm,0.23cm)
          {shared deadline $d_0{=}d_1{=}9$ ticks};
  \end{scope}

\end{tikzpicture}
\caption{%
  Routing decision heatmaps for Experiment~2
  ($\text{tick\_rate}=3$,\;
   $\boldsymbol{\lambda}=(0.25,0.25)$,\;
   $\mathbf{c}=(100,300)$,\;
   $D_0=D_1=3$ physical time units, $d_0=d_1=9$ ticks).
  Each cell $(i,j)$ shows which job type the server assigns when
  $\mathrm{FIL}_0=i$ (vertical axis, $0$ at bottom) and
  $\mathrm{FIL}_1=j$ (horizontal axis, $0$ at left).
  \textbf{Blue}: type~0 served.\quad\textbf{Orange}: type~1 served.
  The dashed red lines mark the shared deadline $d=9$ ticks for both types.
  \textbf{FIFO} (left) splits on the diagonal:
  blue on/below ($\mathrm{FIL}_0 \ge \mathrm{FIL}_1$), orange above---it
  is cost-blind and serves jobs purely by seniority.
  \textbf{RVI} (right) is nearly uniformly orange: the $3\times$ cost advantage
  of type~1 ($c_1=300$ vs.\ $c_0=100$) is large enough that the optimal policy
  \emph{always} prioritises type~1, regardless of which type has been waiting longer.
  The single blue cell at $(\mathrm{FIL}_0{=}10,\,\mathrm{FIL}_1{=}0)$ is the
  unique exception: type~0 has already breached its deadline ($\mathrm{FIL}_0>9$)
  while type~1 has just arrived ($\mathrm{FIL}_1=0$), making it optimal to
  relieve type~0's accruing penalty immediately.
  The resulting FIFO/RVI cost ratio is $1.73$ (72.6\% gap).%
}
\label{fig:heatmap_exp2}
\end{figure}
